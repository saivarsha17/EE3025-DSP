\section{Problem}

\begin{enumerate}[label=\thesection.\arabic*.,ref=\thesection.\theenumi]
    \numberwithin{equation}{enumi}
    
    \item The command     
\begin{lstlisting}
Output_signa1 =  signal.lfilter(b,a,input_signal)

\end{lstlisting}

\item can be executed using  following difference equation
\begin{align}
     \sum _{m=0}^{M}a\brak{m}y\brak{n-m}=\sum _{k=0}^{N}b\brak{k}x\brak{n-k}
\end{align}
 where $x\brak{n} ,y\brak{n}$ are input and  output signals
\end{enumerate}

\section{Solution}
\begin{enumerate}[label=\thesection.\arabic*.,ref=\thesection.\theenumi]
    \numberwithin{equation}{enumi}
\item Z transform of $x(n-k)$ is
\begin{align}
    {\mathcal {Z}\{x(n-k)\} = z^{-k}X(z)}
 \end{align}
Z transform of $y(n-m)$
 \begin{align}
    {\mathcal {Z}\{y(n-m)\} = z^{-m}Y(z)}
 \end{align}
 So, applying Z transform on both sides,


\begin{align}
\sum _{m=0}^{M}a\brak{m}Y\brak{z}z^{-m}=\sum _{k=0}^{N}b\brak{k}X\brak{z}z^{-k}
\end{align}
\begin{align}
    H\brak{z} = \frac{Y\brak{z}}{X\brak{z}} = \frac{\sum _{k=0}^{N}b\brak{k}z^{-k}}{\sum _{m=0}^{M}a\brak{k}z^{-m}}
\end{align}
 We know a  and b coefficients from passing x(n) through low pass butterworth filter.
\item Substitute,
\begin{align}
    z = e^{j\omega}
\end{align}
Where,
\begin{align}
    \omega =\frac{2\pi i}{N}
\end{align}

\begin{align}
    Y\brak{e^{j\omega}} = H\brak{e^{j\omega}}X\brak{e^{j\omega}}
\end{align}
%
Calculate DFT of $y(n)$ and use np.fft.ifft to calculate $y(n)$.
